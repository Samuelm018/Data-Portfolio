2.Identify the highest-rated category in each branch, displaying the branch, category and AVG Rating

SELECT *
FROM (
    SELECT
        branch,
        category,
        avg_rating,
        RANK() OVER (PARTITION BY branch ORDER BY avg_rating DESC) AS rnk
    FROM (
        SELECT
            branch,
            category,
            AVG(rating) AS avg_rating
        FROM walmart
        GROUP BY branch, category
    ) t1
) t2
WHERE rnk = 1;


3.Identify the busiest day for each branch based on the number of transactions

SELECT *
FROM (
    SELECT
        branch,
        day_name,
        no_transactions,
        RANK() OVER (PARTITION BY branch ORDER BY no_transactions DESC) AS rnk
    FROM (
        SELECT
            branch,
            DAYNAME(STR_TO_DATE(`date`, '%d/%m/%y')) AS day_name,
            COUNT(*) AS no_transactions
        FROM walmart
        GROUP BY
            branch,
            DAYNAME(STR_TO_DATE(`date`, '%d/%m/%y'))
    ) t1
) t2
WHERE rnk = 1;

8.Categorize sales into 3 group MORNING, AFTERNOON, EVENING
Find out each of the shift and number of invoices.

SELECT
    branch, 
    CASE 
        WHEN HOUR(`time`) < 12 THEN 'Morning'
        WHEN HOUR(`time`) BETWEEN 12 AND 17 THEN 'Afternoon'
        ELSE 'Evening'
    END AS day_time,
    COUNT(*) AS total_transactions
FROM walmart
GROUP BY branch, day_time
ORDER BY branch, day_time, total_transactions DESC;



9.Identify 5 branch with highest decrease ratio in
revenue compare to last year (current year 2023 and last year 2022)

WITH revenue_2022 AS (
    SELECT 
        branch,
        SUM(total) AS revenue
    FROM walmart
    WHERE YEAR(STR_TO_DATE(`date`, '%d/%m/%Y')) = 2022
    GROUP BY branch
),
revenue_2023 AS (
    SELECT
        branch,
        SUM(total) AS revenue
    FROM walmart
    WHERE YEAR(STR_TO_DATE(`date`, '%d/%m/%Y')) = 2023
    GROUP BY branch
)
SELECT 
    ls.branch,
    ls.revenue AS last_year_revenue,
    cs.revenue AS cr_year_revenue,
    ROUND(
        (ls.revenue - cs.revenue) / ls.revenue * 100,
        2
    ) AS rev_dec_ratio
FROM revenue_2022 ls
JOIN revenue_2023 cs
    ON ls.branch = cs.branch
WHERE ls.revenue > cs.revenue
ORDER BY rev_dec_ratio DESC
LIMIT 5;
